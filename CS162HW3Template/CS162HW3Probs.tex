\begin{enumerate}
\item Let $L$ be the language of all strings of balanced parentheses. That is, all strings of the characters ``('' and ``)'' such that each ``('' has a matching ``)''. Use the Pumping Lemma to show that L is not regular. \newline

\item Let $L = \{0^n 1^{2n} | n > 1 \}$. Show that $L$ is not regular. \newline

\item Given two languages $L$ and $M$, define the exclusive-or of $L$ and $M$ as the set of all strings $w$ such that $w$ is in $L$ and not in $M$ or $w$ is in $M$ and not in $L$. Show that the exclusive-or of two regular languages is regular. \newline

\item Sipser, 1.41. \newline
For languages $A$ and $B$, let the \textbf{perfect shuffle} of $A$ and $B$ be the language
\[\{w \ | \  = a_1 b_1 \ldots a_k b_k, \ \mathrm{where} \ a_1 \ldots a_k \in A \ \mathrm{and} \ b_1 \ldots b_k \in B, \ \mathrm{each} \ a_i, b_i \in \Sigma^*\}\]
Show that the class of regular languages is closed under perfect shuffle. \newline

\item Sipser, 1.70. \newline
We define the \textbf{avoids} operation for languages $A$ and $B$ to be \newline
$A \ \mathrm{avoids} \ B = \{w \ | \ w \in A$ and $w$ doesn't contain any string in $B$ as a substring$\}$. \newline
Prove that the class of regular languages is closed under the avoids operation.
\end{enumerate}
